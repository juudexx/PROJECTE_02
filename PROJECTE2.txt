#comprovar que le usuario es root

        #echo "${USER_NAME}:${PASSWORD}" | chpasswd

        # Test if the command succeeded.
echo "passwd: Success"

        else
            echo "La teva contrasenya es ${PASSWORD}"
            # Cambiar el password el primer cop.
